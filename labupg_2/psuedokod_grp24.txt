FUNCTION SWITCH_NODE(POINTER_TO_NODE)

    IF POINTER_TO_NODE IS NULL
        RETURN

    LET A = VALUE AT POINTER_TO_NODE
    IF A IS NULL OR A->NEXT IS NULL
        RETURN

    LET B = A->NEXT

    LET PREV = A->PREVIOUS
    LET NEXT = B->NEXT

    // CONNECT PREV -> B
    IF PREV EXISTS
        PREV->NEXT = B
    B->PREVIOUS = PREV

    // CONNECT A -> NEXT
    IF NEXT EXISTS
        NEXT->PREVIOUS = A
    A->NEXT = NEXT

    // CONNECT B -> A
    B->NEXT = A
    A->PREVIOUS = B

    // UPDATE LIST HEAD IF A WAS FIRST NODE
    POINTER_TO_NODE = B

END FUNCTION
